** deft
#+begin_src elisp
  (use-package deft
    :ensure t
    :bind ("<f8>" . deft)
    :commands deft
    :init
    (setq
     deft-directory "~/org/"
     deft-recursive t
     ;; de-couples filename and note title:
     deft-use-filename-as-title nil
     deft-use-filter-string-for-filename t
     ;; disable auto-save
     deft-auto-save-interval -1.0
     ;; converts the filter string into a readable file-name using kebab-case:
     deft-file-naming-rules
     '((noslash . "-")
       (nospace . "-")
       (case-fn . downcase)))
    )
      (defun cm/deft-parse-title (file contents)
      "Parse the given FILE and CONTENTS and determine the title.
    If `deft-use-filename-as-title' is nil, the title is taken to
    be the first non-empty line of the FILE.  Else the base name of the FILE is
    used as title."
        (let ((begin (string-match "^#\\+[tT][iI][tT][lL][eE]: .*$" contents)))
          (if begin
              (string-trim (substring contents begin (match-end 0)) "#\\+[tT][iI][tT][lL][eE]: *" "[\n\t ]+")
            (deft-base-filename file))))

      (advice-add 'deft-parse-title :override #'cm/deft-parse-title)

      (setq deft-strip-summary-regexp
            (concat "\\("
                    "[\n\t]" ;; blank
                    "\\|^#\\+[[:alpha:]_]+:.*$" ;; org-mode metadata
                    "\\|^:PROPERTIES:\n\\(.+\n\\)+:END:\n"
                    "\\)"))

#+end_src

  (use-package org-roam
    :ensure t

    (org-roam-directory (file-truename "~/org/"))

           ("C-c n l" . org-roam-buffer-toggle)

           ("C-c n c" . org-roam-capture)
           ("C-c n o" . org-id-get-create)
           ("C-c n t" . org-roam-tag-add)
           ("C-c n a" . org-roam-alias-add)
           ("C-c n r" . org-roam-ref-add)
           ;; Dailies
           ("C-c n j" . org-roam-dailies-capture-today)
           )
    :hook
    (org-roam-mode . visual-line-mode)
    ;; (org-capture-mode . org-align-all-tags)
    :init
    (setq org-roam-v2-ack t)
    ;; (setq org-roam-dailies-directory "daily/")
    :config
    (setq orb-preformat-keywords '("citekey" "title" "url" "file" "author-or-editor" "keywords" "doi" "date" "year" "journal" "journaltitle"))
    (setq org-roam-capture-templates
     '(
       ("d" "default" plain "%?" 
        :if-new (file+head "%<%Y%m%d%H%M%S>-${slug}.org" "#+title: ${title}")
        :unnarrowed t)
       ("a" "article" plain "* Citation:\n* Notes:\n- "
        :if-new (file+head "articles/${citekey}.org" "#+title: ${title}\n#+filetags: source_note:article\n")
        :unnarrowed t)
       ("m" "meeting" plain "* Attendees:\n* Agenda:\n* Notes:\n"
        :if-new (file+head "meetings/%^{Enter meeting dat yyyymmddhhmm}-${slug}.org" "#+title: ${title}\n#+filetags: meeting\n")
        :unnarrowed t)
       ("p" "person" plain "* Interests:\n- \n* Notes:\n- %?" 
        :if-new (file+head "people/%<%Y%m%d%H%M%S>-${slug}.org" "#+title: ${title}\n#+filetags: person\n")
        :unnarrowed t)
      )
    )
    (setq org-roam-capture-ref-templates
          '(("r" "ref" plain "%?"
             :target (file+head "webpages/%<%Y%m%d%H%M>-${slug}.org" "#+title: ${title}\n#+filetags: source_note:webpage")
             :unnarrowed t)))
    (org-roam-db-autosync-mode)
    ;; If using org-roam-protocol
    (require 'org-roam-protocol))
    (add-to-list 'display-buffer-alist
               '("\\*org-roam\\*"
                 (display-buffer-in-direction)
                 (direction . right)
                 (window-width . 0.33)
                 (window-height . fit-window-to-buffer)))

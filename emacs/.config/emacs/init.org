* Startup Reporting

Add a hook that reports how long and how many garbage collections the
startup took. The hook is run at the very end, so the message doesn't
get clobbered by other messages during startup.

#+begin_src emacs-lisp
  (add-hook
   'emacs-startup-hook
   (lambda ()
   (setq gc-cons-threshold (* 16 1024 1024))
     (message
      "Emacs ready in %s with %d garbage collections."
      (format "%.2f seconds"
              (float-time
               (time-subtract after-init-time before-init-time)))
      gcs-done)))
#+end_src

Bypass the ~org-element-cache-map~ call in ~org-get-buffer-tags~ which
causes a freeze in large buffers by replacing it with a fast regexp scan.

#+begin_src emacs-lisp :tangle no :noweb-ref org-config
  (advice-add 'org-get-buffer-tags :override
              (lambda ()
                (let (tags)
                  (org-with-wide-buffer
                   (goto-char (point-min))
                   (while (re-search-forward
                           "^\\*+.*[ \t]\\(:[[:alnum:]_@#%:]+:\\)[ \t]*$"
                           nil t)
                     (dolist (tag (split-string
                                   (match-string-no-properties 1) ":" t))
                       (cl-pushnew (list tag) tags :test #'equal))))
                  tags)))
#+end_src

(defun jdl/refile-to-notes-file ()
  "Refile subtree at point to the top level of a file in the notes directory."
  (interactive)
  (let* ((dir (expand-file-name vulpea-default-notes-directory))
         (files (directory-files dir t "\\.org$"))
         (target (completing-read "Refile to: "
                                  (mapcar #'file-name-nondirectory files)
                                  nil t)))
    (org-cut-subtree)
    (with-current-buffer (find-file-noselect (expand-file-name target dir))
      (goto-char (point-max))
      (unless (bolp) (insert "\n"))
      (org-paste-subtree 1)
      (save-buffer))
    (message "Refiled to %s" target)))
#+end_src

#+begin_src emacs-lisp
  (defun make-meeting-template (key name file company-id)
    "Create a meeting capture template for the given key, name, file, and company ID."
    (list key name 'entry
          `(file+olp+datetree ,file)
          (string-join
           (list (concat "* %^{Meeting Title?}"  " :meeting:")
                  ":PROPERTIES:"
                  (format org-property-format ":ID:" (org-id-uuid))
                  (format org-property-format ":CREATED:"
                          (format-time-string "%Y-%m-%dT%H:%M:%S"))
                  ":END:"
                  "- date :: %T"
                  "- attendee :: John Lord"
                  (concat "- company :: [[id:" company-id "][" name "]]")
                  "- attendee :: "
                  ""
                  "** Agenda"
                  "*** Subject 1"

=jdl/vulpea-extract-subtree= is the vulpea equivalent of =org-roam-extract-subtree=. It promotes the heading at point to a standalone vulpea note: the heading becomes the =#+title=, any tags on the heading become =#+filetags=, and the subtree body (including all subheadings) is carried across with heading levels promoted so they read naturally at the top level. The original subtree is removed and replaced with an =[[id:...]]=  link to the new file. =jdl/vulpea--promote-body= is the private helper that reduces all heading star counts in a string by a given amount.

The body is captured by snapshotting the raw subtree text while point is pinned at the target heading (via an explicit =org-back-to-heading= call), then stripping the heading line and property drawer in a temp buffer. This avoids a subtle bug where =org-id-get-create= — called earlier in the =let*= chain — can leave point displaced, causing =org-end-of-meta-data='s internal =org-back-to-heading= to snap to the first subheading rather than the target heading, which made only the first subheading's content appear in the extracted file.

After the subtree is cut and the new file written, =org-id-add-location= is called to register the ID against the new file in =org-id-locations=. Without this, =org-id-get-create= leaves the ID registered against the source file; after =org-cut-subtree= removes the heading, vulpea and =id:= links cannot resolve it.

#+begin_src emacs-lisp
(defun jdl/vulpea--promote-body (content level)
  "Return CONTENT string with all org heading levels reduced by LEVEL."
  (with-temp-buffer
    (insert content)
    (goto-char (point-min))
    (while (re-search-forward "^\\(\\*+\\)" nil t)
      (let* ((n   (length (match-string 1)))
             (new (make-string (max 1 (- n level)) ?*)))
        (replace-match new nil nil nil 1)))
    (buffer-string)))

(defun jdl/vulpea-extract-subtree ()
  "Promote the heading at point to a standalone vulpea note file.
Creates a new .org file in `vulpea-default-notes-directory' with the
heading as title, the heading's tags as #+filetags, and the subtree
body (with heading levels promoted) as note content.  The subtree is
removed from the current buffer and replaced with an [[id:...]] link."
  (interactive)
  (org-back-to-heading t)
  (let* ((title  (org-get-heading t t t t))
         (tags   (org-get-tags nil t))
         (id     (org-id-get-create))
         (level  (org-current-level))
         ;; Capture body: grab the full raw subtree text while point is
         ;; guaranteed at the heading (org-id-get-create may have moved it),
         ;; then strip the heading line and property drawer in a temp buffer
         ;; so that org-end-of-meta-data and org-back-to-heading always
         ;; operate on predictable content with no risk of snapping to a
         ;; subheading in the live buffer.
         (body   (save-excursion
                   (org-back-to-heading t)
                   (let* ((raw-end (save-excursion (org-end-of-subtree t) (point)))
                          (raw     (buffer-substring-no-properties (point) raw-end)))
                     (with-temp-buffer
                       (org-mode)
                       (insert raw)
                       (goto-char (point-min))
                       (org-end-of-meta-data t)
                       (buffer-substring-no-properties (point) (point-max))))))
         (ts     (format-time-string "%Y%m%d%H%M%S"))
         (slug   (replace-regexp-in-string
                  "[^a-z0-9]+" "_"
                  (downcase (string-trim title))))
         (file   (expand-file-name
                  (concat ts "_" slug ".org")
                  (expand-file-name vulpea-default-notes-directory))))
    ;; Write the new vulpea-format file
    (with-temp-file file
      (insert ":PROPERTIES:\n"
              ":ID:       " id "\n"
              ":CREATED:  " (format-time-string "%Y-%m-%dT%H:%M:%S") "\n"
              ":END:\n"
              "#+title: " title "\n")
      (when tags
        (insert "#+filetags: :" (string-join tags ":") ":\n"))
      (insert "#+created: " (format-time-string "[%Y-%m-%d]") "\n")
      (let ((promoted (jdl/vulpea--promote-body body level)))
        (unless (string-empty-p (string-trim promoted))
          (insert "\n" (string-trim promoted) "\n"))))
    ;; Replace the subtree with an id: link and open the new file
    (org-cut-subtree)
    (insert (format "[[id:%s][%s]]\n" id title))
    ;; org-id-get-create registered this ID against the source file.
    ;; Now that the content lives in the new file, update org-id-locations
    ;; so that id: links and vulpea can resolve it correctly.
    (org-id-add-location id (expand-file-name file))
    (vulpea-db-sync)
    (find-file file)))
#+end_src

=jdl/migrate-article-notes= is a one-shot migration command for notes originally created with =citar-org-roam=. It scans all =.org= files in =vulpea-default-notes-directory=, and for each file that has a =:ROAM_REFS:= property but no =- citekey ::= meta line it performs four changes: adds =:CITEKEY:= to the property drawer (so org-mode can find it), inserts =- citekey ::= as the first meta line (so the vulpea db query can match it), changes the =:source:= filetag to =:article:=, and renames =- year ::= to =- date ::=. Filenames and note content are left untouched. Run it once with =M-x jdl/migrate-article-notes= then call =M-x vulpea-db-sync= to reindex.

#+begin_src emacs-lisp
(defun jdl/migrate-article-notes ()
  "Migrate citar-org-roam article notes to vulpea format.
For each .org file in `vulpea-default-notes-directory' that has a
:ROAM_REFS: [cite:@KEY] property but no '- citekey ::' meta line:
  - Adds :CITEKEY: KEY to the property drawer
  - Inserts '- citekey :: KEY' as the first meta line
  - Changes #+filetags: :source: to :article:
  - Renames '- year ::' to '- date ::'
Run M-x vulpea-db-sync afterwards to reindex."
  (interactive)
  (let ((notes-dir (expand-file-name vulpea-default-notes-directory))
        (migrated '()))
    (dolist (file (directory-files notes-dir t "\\.org$"))
      (with-current-buffer (find-file-noselect file)
        (save-excursion
          (goto-char (point-min))
          ;; Skip files that already have a citekey meta line
          (unless (re-search-forward "^- citekey ::" nil t)
            (goto-char (point-min))
            ;; Extract citekey from :ROAM_REFS: [cite:@KEY]
            (when (re-search-forward
                   ":ROAM_REFS:[ \t]+\\[cite:@\\([^]]+\\)\\]" nil t)
              (let ((citekey (match-string 1)))
                ;; 1. Add :CITEKEY: to property drawer (before :END:)
                (goto-char (point-min))
                (when (re-search-forward "^:END:" nil t)
                  (beginning-of-line)
                  (insert (format ":CITEKEY:  %s\n" citekey)))
                ;; 2. Insert '- citekey ::' after #+filetags (or #+title)
                (goto-char (point-min))
                (if (re-search-forward "^#\\+filetags:.*$" nil t)
                    (end-of-line)
                  (re-search-forward "^#\\+title:.*$" nil t)
                  (end-of-line))
                (insert (format "\n- citekey :: %s" citekey))
                ;; 3. Change :source: tag to :article:
                (goto-char (point-min))
                (when (re-search-forward "^\\(#\\+filetags:.*\\):source:\\(.*\\)$" nil t)
                  (replace-match "\\1:article:\\2"))
                ;; 4. Rename '- year ::' to '- date ::'
                (goto-char (point-min))
                (while (re-search-forward "^- year ::" nil t)
                  (replace-match "- date ::"))
                (save-buffer)
                (push (file-name-nondirectory file) migrated)))))))
    (if migrated
        (message "Migrated %d note(s):\n%s"
                 (length migrated)
                 (mapconcat #'identity (sort migrated #'string<) "\n"))
      (message "No notes needed migration."))))
#+end_src

             (title  (or (jdl/citar-entry-get "title"        entry) key))
             (authors (mapcar #'string-trim
                              (split-string
                               (or (jdl/citar-entry-get "author" entry) "")
                               " and " t)))
             (date   (or (jdl/citar-entry-get "date"         entry)
                         (jdl/citar-entry-get "year"         entry) ""))
             (doi    (let ((d (jdl/citar-entry-get "doi" entry)))
                       (if d (format "[[https://doi.org/%s][%s]]" d d) "")))
             (url    (or (jdl/citar-entry-get "url"          entry) ""))
             (journal (or (jdl/citar-entry-get "journaltitle" entry)
                          (jdl/citar-entry-get "journal"      entry) ""))
             (note (let ((vulpea-create-default-function
                         (lambda (_title)
                           (list :file-name (concat "${timestamp}_" key ".org")))))
                     (vulpea-create
                      title
                      nil
                      :tags '("article")
                      :head "#+created: %<[%Y-%m-%d]>"
                      :properties `(("CREATED" . "%<[%Y-%m-%d]>")
                                    ("CITEKEY" . ,key))
                      :meta `(("citekey" . ,key)
                              ,@(mapcar (lambda (a) (cons "author" a)) authors)
                              ("date" . ,date)
                              ("journal" . ,journal)
                              ("doi" . ,doi)
                              ("url" . ,url))
                      :body "* Notes\n\n* References\n"))))
        (find-file (vulpea-note-path note))))))

(defun jdl/vulpea-capture-article-from-citar ()
  "Select a reference and open or create a vulpea note for it."
  (interactive)
  (jdl/vulpea-open-or-create-article (citar-select-ref)))

(defun jdl/citar-vulpea-open (key)
  "Open a vulpea note for KEY from citar/embark.
When a note already exists, citar passes its file path; otherwise KEY
is the citekey and we create a new note."
  (if (file-exists-p (expand-file-name key))
      (find-file key)
    (jdl/vulpea-open-or-create-article key)))

(defun jdl/citar-vulpea-get-notes (&optional keys)
  "Return a hash table mapping citekeys to lists of vulpea note paths.
If KEYS is non-nil, restrict results to those keys."
  (let ((result (make-hash-table :test #'equal))
        (notes (vulpea-db-query
                (lambda (note)
                  (vulpea-note-meta-get note "citekey" 'string)))))
    (dolist (note notes)
      (let ((key (vulpea-note-meta-get note "citekey" 'string)))
        (when (or (null keys) (member key keys))
          (puthash key
                   (cons (vulpea-note-path note)
                         (gethash key result nil))
                   result))))
    result))

(defun jdl/citar-vulpea-has-notes ()
  "Return a predicate checking whether a citekey has a vulpea note.
Citar calls this with no arguments and calls the result with each citekey."
  (let ((keys (delq nil
                    (mapcar (lambda (note)
                              (vulpea-note-meta-get note "citekey" 'string))
                            (vulpea-db-query
                             (lambda (note)
                               (vulpea-note-meta-get note "citekey" 'string)))))))
    (lambda (key) (member key keys))))

#+end_src
